    <script>
        const kanbanData = [
            { id: 1, column: 'backlog', title: 'Global Regulatory Sync', desc: 'Synthesizing the unified 2026 safety standards across EU, US, and Asian markets.', priority: 'high' },
            { id: 2, column: 'backlog', title: 'Zero-HEMA Sourcing', desc: 'Securing plant-derived acrylate alternatives for 2026 formulations.', priority: 'high' },
            { id: 3, column: 'inprogress', title: 'AI-Photon Lab', desc: 'Developing smart-cure LED hardware that adjusts to gel thickness.', priority: 'high' },
            { id: 4, column: 'inprogress', title: 'Omnichannel Kit', desc: 'Premium DIY system for US market launch.', priority: 'medium' },
            { id: 5, column: 'testing', title: 'Bio-Stability Trials', desc: '48-month shelf-life validation for the 2026 bio-gel line.', priority: 'high' },
            { id: 6, column: 'done', title: 'HEMA Zero Standard', desc: 'Successfully certified the first 0.0% HEMA-Free line for launch.', priority: 'none' }
        ];

        const columns = [
            { id: 'backlog', title: 'Discovery' },
            { id: 'inprogress', title: 'Engineering' },
            { id: 'testing', title: 'Validation' },
            { id: 'done', title: 'Launch' }
        ];

        function getPriorityColor(p) {
            switch(p) {
                case 'high': return '#FBCFE8'; 
                case 'medium': return '#BAE6FD'; 
                default: return '#F1F5F9';
            }
        }

        function renderKanban(filter = 'all') {
            const board = document.getElementById('kanban-board');
            board.innerHTML = '';
            columns.forEach(col => {
                const colDiv = document.createElement('div');
                colDiv.className = 'kanban-column flex flex-col space-y-6';
                colDiv.innerHTML = `<h3 class="text-xl font-bold px-2">${col.title}</h3>`;
                const container = document.createElement('div');
                container.className = 'space-y-4';
                const filtered = kanbanData.filter(i => i.column === col.id && (filter === 'all' || i.priority === filter));
                filtered.forEach(item => {
                    const card = document.createElement('div');
                    card.className = `glass-card p-6 rounded-[2rem] cursor-pointer`;
                    card.style.borderLeft = `8px solid ${getPriorityColor(item.priority)}`;
                    card.onclick = () => openModal(item);
                    card.innerHTML = `<h4 class="font-bold text-slate-700 text-sm">${item.title}</h4>`;
                    container.appendChild(card);
                });
                colDiv.appendChild(container);
                board.appendChild(colDiv);
            });
        }

        function openModal(item) {
            document.getElementById('modal-title').textContent = item.title;
            document.getElementById('modal-desc').textContent = item.desc;
            document.getElementById('card-modal').classList.add('active');
        }
        function closeModal() { document.getElementById('card-modal').classList.remove('active'); }

        document.getElementById('kanban-filter').onchange = (e) => renderKanban(e.target.value);
        function scrollToSection(id) { document.getElementById(id).scrollIntoView({ behavior: 'smooth' }); }

        function initChart() {
            const ctx = document.getElementById('resourceChart').getContext('2d');
            new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: ['R&D', 'Safety', 'Marketing', 'Logistics'],
                    datasets: [{
                        data: [45, 25, 20, 10],
                        backgroundColor: ['#FBCFE8', '#BAE6FD', '#E9D5FF', '#F1F5F9'],
                        borderWidth: 5,
                        borderColor: '#ffffff',
                        borderRadius: 15
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    cutout: '75%',
                    plugins: { legend: { position: 'bottom', labels: { padding: 20, usePointStyle: true, font: { weight: 'bold' } } } }
                }
            });
        }

        window.onload = () => { renderKanban(); initChart(); };
    </script>
